1. Patching Admin Track Replacement Vulnerabilities
Your dashboard flags two distinct issues when an admin replaces or updates scripture audio tracks:

Memory Buffering Abuse: Large audio files are buffered completely in memory before checking if the user actually has admin privileges.

Stored XSS via File Origin: Attackers uploading malicious HTML/SVG files masquerading as audio, which then execute script code under your main domain.

The Secure Admin Route (/api/admin/audio/route.ts)
To fix both, we check the user's role before parsing the incoming file stream, enforce strict MIME-type validation, and offload the stream safely.

TypeScript
import { NextResponse } from 'next/server';
import { getAuth } from '@clerk/nextjs/server';
import { prisma } from '@/lib/db';

// Allowed audio types only — strict block on SVG/HTML or ambiguous blobs
const ALLOWED_MIME_TYPES = ['audio/mpeg', 'audio/mp3', 'audio/wav', 'audio/ogg', 'audio/aac'];

export async function POST(request: Request) {
  // 1. IMMEDIATE AUTH CHECK (Fixes buffering before privilege verification)
  const { userId } = getAuth(request as any);
  if (!userId) {
    return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
  }

  const user = await prisma.user.findUnique({ where: { id: userId } });
  if (!user || user.role !== 'ADMIN') {
    return NextResponse.json({ error: "Forbidden: Admin access required" }, { status: 403 });
  }

  try {
    // 2. STREAM VALIDATION (Fixes memory exhaustion & XSS)
    const formData = await request.formData();
    const file = formData.get('audioFile') as File;

    if (!file) {
      return NextResponse.json({ error: "No file provided" }, { status: 400 });
    }

    if (!ALLOWED_MIME_TYPES.includes(file.type)) {
      return NextResponse.json({ 
        error: "Security Violation: Only valid audio files (MP3, WAV, OGG, AAC) are permitted." 
      }, { status: 400 });
    }

    // Process the validated stream smoothly to your storage bucket (S3, Cloudflare R2, etc.)
    // const audioUrl = await uploadToStorageBucket(file.stream());

    return NextResponse.json({ success: true, message: "Audio track updated safely." });
  } catch (error) {
    return NextResponse.json({ error: "Processing failed" }, { status: 500 });
  }
}

import React from 'react';
import { View, Text, FlatList, TouchableOpacity } from 'react-native';

const THEMATIC_PATHS = [
  { id: '1', title: 'Finding Peace in Anxiety', count: '12 Verses', faiths: ['Gita', 'Bible', 'Tao Te Ching'] },

  { id: '3', title: 'Cultivating Gratitude', count: '10 Verses', faiths: ['Torah', 'Gita', 'Dhammapada'] },
];

export function ThematicPathsScreen({ navigation }) {
  return (
    <View style={{ flex: 1, backgroundColor: '#111', padding: 20 }}>
      <Text style={{ fontSize: 24, fontWeight: 'bold', color: '#fff', marginBottom: 4 }}>
        Choose Your Path
      </Text>
      <Text style={{ fontSize: 14, color: '#aaa', marginBottom: 20 }}>

      </Text>

      <FlatList

        keyExtractor={(item) => item.id}
        renderItem={({ item }) => (
